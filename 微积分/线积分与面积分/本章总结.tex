\section{本章总结}
梯度公式\[
	\grad f
	= \begin{bmatrix}
		\pdv{f}{x} & \pdv{f}{y} & \pdv{f}{z}
	\end{bmatrix}^T.
\]

散度公式\[
	\div{\vb{A}}
	= \pdv{P}{x}+\pdv{Q}{y}+\pdv{R}{z}.
\]

旋度公式\[
	\curl{\vb{A}}
	= \begin{vmatrix}
		\vb{i} & \vb{j} & \vb{k} \\
		\pdv{x} & \pdv{y} & \pdv{z} \\
		P & Q & R \\
	\end{vmatrix}.
\]

高斯公式\[
	\iiint_\Omega \div{\vb{A}} \dd{v}
	= \varoiint_\Sigma \vb{A} \cdot \dd{\vb{S}}.
\]

斯托克斯公式\[
	\iint_\Sigma \curl{\vb{A}} \cdot \dd{\vb{S}}
	= \oint_L \vb{A} \cdot \dd{\vb{s}}.
\]
