\section{本章总结}
\begin{gather*}
	\int_0^{\frac{\pi}{2}} \sin^n x \dd{x}
	= \int_0^{\frac{\pi}{2}} \cos^n x \dd{x}
	= \left\{ \def\arraystretch{1.5} \begin{array}{rl}
		\frac{\pi}{2}\frac{(n-1)!!}{n!!},
			& \text{$n$是偶数}, \\
		\frac{(n-1)!!}{n!!},
			& \text{$n$是奇数}.
	\end{array} \right.
\end{gather*}
